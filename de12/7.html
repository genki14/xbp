利用ソースコード

  1. import pandas as pd # データ分析に用いるライブラリ
  2. import matplotlib.pyplot as plt # グラフ表示に用いるライブラリ
  4. pd.set_option('display.unicode.east_asian_width', True) # 表示のずれを少し緩和
  5. plt.rcParams['font.family'] = 'IPAexGothic' # グラフ表示におけるフォントの指定
  6. plt.rcParams['font.size'] = 8
  8. data_path = "./data.csv"
  9. df_data = pd.read_csv(data_path, encoding="utf-8-sig")
  11. # 駅名を入力で合計時間、家賃、面積、築年数の平均値を算出
  12. print(df_data.groupby("駅").mean().loc["大船駅", ["合計時間", "家賃", "面積", "築年数"]])
  13. # 路線を入力で合計時間、家賃、面積、築年数の平均を各駅ごとに並べた棒グラフを表示
  14. df_mean = df_data[df_data.loc[:, "路線"] == "湘南新宿ライン高海"]
  15. df_mean.groupby("駅").mean().loc[:, ["合計時間", "家賃", "面積", "築年数"]].plot.bar(subplots=True,
  16.                                                                          layout=(2, 2),
  17.                                                                         figsize=(16, 10))
  18. #グラフの位置調整
  19. plt.subplots_adjust(left=0.05, right=0.99, wspace=0.05, bottom=0.2)
  20. plt.show()
  21. print(df_data.columns)
  22. print(df_data)
